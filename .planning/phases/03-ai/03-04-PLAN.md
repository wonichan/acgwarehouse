// lib/services/tag_service.dart
class TagService {
  final String baseUrl;
  final http.Client client;

  TagService({this.baseUrl = 'http://localhost:8080/api/v1', http.Client? client})
      : client = client ?? http.Client();

  // 获取标签列表
  Future<List<Tag>> fetchTags({String? search, int limit = 50, int offset = 0}) async {
    final uri = Uri.parse('$baseUrl/tags').replace(
      queryParameters: {
        if (search != null) 'search': search,
        'limit': limit.toString(),
        'offset': offset.toString(),
      },
    );
    final response = await client.get(uri);
    if (response.statusCode != 200) throw Exception('Failed to fetch tags');
    final data = jsonDecode(response.body);
    return (data['tags'] as List).map((e) => Tag.fromJson(e)).toList();
  }

  // 搜索标签（支持别名匹配）
  Future<List<Tag>> searchTags(String query) async {
    return fetchTags(search: query);
  }

  // 获取图片的标签
  Future<Map<String, List<Tag>>> getImageTags(int imageId) async {
    final response = await client.get(Uri.parse('$baseUrl/images/$imageId/tags'));
    if (response.statusCode != 200) throw Exception('Failed to fetch image tags');
    final data = jsonDecode(response.body);
    return {
      'confirmed': (data['confirmed'] as List).map((e) => Tag.fromJson(e)).toList(),
      'pending': (data['pending'] as List).map((e) => Tag.fromJson(e)).toList(),
      'rejected': (data['rejected'] as List).map((e) => Tag.fromJson(e)).toList(),
    };
  }

  // 为图片添加标签
  Future<Tag> addImageTag(int imageId, {int? tagId, String? tagLabel}) async {
    final response = await client.post(
      Uri.parse('$baseUrl/images/$imageId/tags'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({
        if (tagId != null) 'tag_id': tagId,
        if (tagLabel != null) 'tag_label': tagLabel,
      }),
    );
    if (response.statusCode != 200 && response.statusCode != 201) {
      throw Exception('Failed to add tag');
    }
    return Tag.fromJson(jsonDecode(response.body));
  }

  // 移除图片标签
  Future<void> removeImageTag(int imageId, int tagId) async {
    final response = await client.delete(
      Uri.parse('$baseUrl/images/$imageId/tags/$tagId'),
    );
    if (response.statusCode != 200) throw Exception('Failed to remove tag');
  }

  // 确认标签
  Future<void> confirmTag(int imageId, int tagId) async {
    final response = await client.post(
      Uri.parse('$baseUrl/images/$imageId/tags/$tagId/review'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({'action': 'confirm'}),
    );
    if (response.statusCode != 200) throw Exception('Failed to confirm tag');
  }

  // 拒绝标签
  Future<void> rejectTag(int imageId, int tagId) async {
    final response = await client.post(
      Uri.parse('$baseUrl/images/$imageId/tags/$tagId/review'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({'action': 'reject'}),
    );
    if (response.statusCode != 200) throw Exception('Failed to reject tag');
  }

  // 批量确认标签
  Future<void> batchConfirmTags(int imageId, List<int> tagIds) async {
    final response = await client.post(
      Uri.parse('$baseUrl/images/$imageId/tags/batch-review'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode({'tag_ids': tagIds, 'action': 'confirm'}),
    );
    if (response.statusCode != 200) throw Exception('Failed to batch confirm');
  }

  // 触发 AI 标签生成
  Future<int> triggerAITags(int imageId) async {
    final response = await client.post(
      Uri.parse('$baseUrl/images/$imageId/ai-tags'),
    );
    if (response.statusCode != 200) throw Exception('Failed to trigger AI tags');
    return jsonDecode(response.body)['job_id'];
  }

  // 获取 AI 任务状态
  Future<Map<String, dynamic>> getAITagStatus(int imageId) async {
    final response = await client.get(
      Uri.parse('$baseUrl/images/$imageId/ai-tags/status'),
    );
    if (response.statusCode != 200) throw Exception('Failed to get AI status');
    return jsonDecode(response.body);
  }
}

// lib/providers/tag_provider.dart
import 'package:flutter/foundation.dart';
import '../models/tag.dart';
import '../services/tag_service.dart';

class TagProvider extends ChangeNotifier {
  final TagService _tagService;

  List<Tag> _allTags = [];
  List<Tag> _filteredTags = [];
  Set<int> _selectedTagIds = {};
  bool _isLoading = false;
  String? _error;

  TagProvider(this._tagService);

  // Getters
  List<Tag> get allTags => _allTags;
  List<Tag> get filteredTags => _filteredTags;
  Set<int> get selectedTagIds => _selectedTagIds;
  bool get isLoading => _isLoading;
  String? get error => _error;
  List<Tag> get selectedTags => _allTags.where((t) => _selectedTagIds.contains(t.id)).toList();

  // 加载所有标签
  Future<void> loadTags() async {
    _isLoading = true;
    _error = null;
    notifyListeners();

    try {
      _allTags = await _tagService.fetchTags();
      _filteredTags = _allTags;
    } catch (e) {
      _error = e.toString();
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }

  // 搜索标签
  Future<void> searchTags(String query) async {
    if (query.isEmpty) {
      _filteredTags = _allTags;
    } else {
      _filteredTags = await _tagService.searchTags(query);
    }
    notifyListeners();
  }

  // 选择/取消选择标签
  void toggleTag(int tagId) {
    if (_selectedTagIds.contains(tagId)) {
      _selectedTagIds.remove(tagId);
    } else {
      _selectedTagIds.add(tagId);
    }
    notifyListeners();
  }

  // 清空选择
  void clearSelection() {
    _selectedTagIds.clear();
    notifyListeners();
  }

  // 确认图片标签
  Future<void> confirmImageTag(int imageId, int tagId) async {
    await _tagService.confirmTag(imageId, tagId);
    notifyListeners();
  }

  // 拒绝图片标签
  Future<void> rejectImageTag(int imageId, int tagId) async {
    await _tagService.rejectTag(imageId, tagId);
    notifyListeners();
  }
}

              },

import 'package:flutter/material.dart';
import '../models/tag.dart';

enum TagChipStyle {
  confirmed,   // 绿色，已确认
  pending,     // 橙色，待确认
  rejected,    // 红色，已拒绝
  neutral,     // 默认灰色
}

class TagChip extends StatelessWidget {
  final Tag tag;
  final TagChipStyle style;
  final VoidCallback? onTap;
  final VoidCallback? onConfirm;
  final VoidCallback? onReject;
  final VoidCallback? onDelete;
  final bool showActions;

  const TagChip({
    Key? key,
    required this.tag,
    this.style = TagChipStyle.neutral,
    this.onTap,
    this.onConfirm,
    this.onReject,
    this.onDelete,
    this.showActions = false,
  }) : super(key: key);

  Color get _backgroundColor {
    switch (style) {
      case TagChipStyle.confirmed:
        return Colors.green.shade100;
      case TagChipStyle.pending:
        return Colors.orange.shade100;
      case TagChipStyle.rejected:
        return Colors.red.shade100;
      default:
        return Colors.grey.shade200;
    }
  }

  Color get _textColor {
    switch (style) {
      case TagChipStyle.confirmed:
        return Colors.green.shade800;
      case TagChipStyle.pending:
        return Colors.orange.shade800;
      case TagChipStyle.rejected:
        return Colors.red.shade800;
      default:
        return Colors.grey.shade800;
    }
  }

  @override
  Widget build(BuildContext context) {
    return GestureDetector(
      onTap: onTap,
      child: Container(
        margin: EdgeInsets.symmetric(horizontal: 4, vertical: 2),
        padding: EdgeInsets.symmetric(horizontal: 8, vertical: 4),
        decoration: BoxDecoration(
          color: _backgroundColor,
          borderRadius: BorderRadius.circular(16),
        ),
        child: Row(
          mainAxisSize: MainAxisSize.min,
          children: [
            Text(
              tag.preferredLabel,
              style: TextStyle(color: _textColor, fontSize: 14),
            ),
            if (showActions) ...[
              SizedBox(width: 4),
              if (onConfirm != null)
                InkWell(
                  onTap: onConfirm,
                  child: Icon(Icons.check, size: 16, color: Colors.green),
                ),
              if (onReject != null)
                InkWell(
                  onTap: onReject,
                  child: Icon(Icons.close, size: 16, color: Colors.red),
                ),
              if (onDelete != null)
                InkWell(
                  onTap: onDelete,
                  child: Icon(Icons.delete_outline, size: 16, color: Colors.grey),
                ),
            ],
          ],
        ),
      ),
    );
  }
}

class _ImageDetailScreenState extends State<ImageDetailScreen> {

class AddTagDialog extends StatefulWidget {
  final int imageId;
  const AddTagDialog({Key? key, required this.imageId}) : super(key: key);

  @override
  State<AddTagDialog> createState() => _AddTagDialogState();
}

class _AddTagDialogState extends State<AddTagDialog> {
  final _controller = TextEditingController();
  List<Tag> _suggestions = [];
  bool _loading = false;

  @override
  Widget build(BuildContext context) {
    return AlertDialog(
      title: Text('添加标签'),
      content: Column(
        mainAxisSize: MainAxisSize.min,
        children: [
          TextField(
            controller: _controller,
            decoration: InputDecoration(hintText: '输入标签名称'),
            onChanged: _searchTags,
          ),
          if (_loading) CircularProgressIndicator(),
          if (_suggestions.isNotEmpty)
            Container(
              height: 200,
              child: ListView.builder(
                shrinkWrap: true,
                itemCount: _suggestions.length,
                itemBuilder: (context, index) {
                  final tag = _suggestions[index];
                  return ListTile(
                    title: Text(tag.preferredLabel),
                    onTap: () => _selectTag(tag.id),
                  );
                },
              ),
            ),
        ],
      ),
      actions: [
        TextButton(child: Text('取消'), onPressed: () => Navigator.pop(context)),
        ElevatedButton(
          child: Text('创建新标签'),
          onPressed: () => _createNewTag(_controller.text),
        ),
      ],
    );
  }

  Future<void> _searchTags(String query) async {
    if (query.isEmpty) {
      setState(() => _suggestions = []);
      return;
    }
    setState(() => _loading = true);
    final tagService = context.read<TagService>();
    final tags = await tagService.searchTags(query);
    setState(() {
      _suggestions = tags;
      _loading = false;
    });
  }

  Future<void> _selectTag(int tagId) async {
    final tagService = context.read<TagService>();
    await tagService.addImageTag(widget.imageId, tagId: tagId);
    Navigator.pop(context);
  }

  Future<void> _createNewTag(String label) async {
    if (label.isEmpty) return;
    final tagService = context.read<TagService>();
    await tagService.addImageTag(widget.imageId, tagLabel: label);
    Navigator.pop(context);
  }
}
