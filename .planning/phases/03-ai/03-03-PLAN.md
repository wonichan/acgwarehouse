API 端点：

1. `GET /api/v1/tags` - 获取标签列表
   - Query: search, limit, offset
   - Response: `{ "tags": [...], "total": N }`
   - 按 usage_count DESC 排序

2. `POST /api/v1/tags` - 创建标签
   - Body: `{ "preferred_label": "蓝天", "primary_category": "场景" }`
   - Response: `{ "id": 1, "preferred_label": "蓝天", ... }`
   - 自动生成 slug

3. `PUT /api/v1/tags/:id` - 更新标签
   - Body: `{ "preferred_label": "蓝色天空" }`
   - Response: 更新后的标签

4. `DELETE /api/v1/tags/:id` - 删除标签
   - 同时删除关联的 image_tags 和 tag_aliases
   - Response: `{ "success": true }`

5. `GET /api/v1/tags/:id/aliases` - 获取标签别名
   - Response: `{ "aliases": [...] }`

6. `POST /api/v1/tags/:id/aliases` - 添加别名
   - Body: `{ "label": "蓝色天空", "alias_type": "synonym" }`
   - Response: 创建的别名对象

7. `DELETE /api/v1/tags/:id/aliases/:alias_id` - 删除别名

搜索逻辑：
- 先搜索 tags.preferred_label
- 再搜索 tag_aliases.label，返回关联的标签
- 合并去重

API 端点：

1. `GET /api/v1/images/:id/tags` - 获取图片标签
   - Query: 无
   - Response:
     ```json
     {
       "confirmed": [...],  // review_state = "confirmed"
       "pending": [...],    // review_state = "pending"
       "rejected": [...]    // review_state = "rejected"
     }
     ```

2. `POST /api/v1/images/:id/tags` - 为图片添加标签
   - Body: `{ "tag_id": 1 }` 或 `{ "tag_label": "新标签" }`
   - 如果 tag_label 不存在，创建新标签并关联
   - 用户手动添加的标签默认 review_state = "confirmed"

3. `DELETE /api/v1/images/:id/tags/:tag_id` - 移除图片标签
   - Response: `{ "success": true }`

4. `POST /api/v1/images/:id/tags/:tag_id/review` - 确认/拒绝标签
   - Body: `{ "action": "confirm" }` 或 `{ "action": "reject" }`
   - 更新 image_tags.review_state
   - Response: `{ "success": true, "review_state": "confirmed" }`

5. `POST /api/v1/images/:id/tags/batch-review` - 批量确认
   - Body: `{ "tag_ids": [1, 2, 3], "action": "confirm" }`
   - 批量更新多个标签状态
   - Response: `{ "success": true, "updated_count": 3 }`

```go
func SetupRoutes(r *gin.Engine, deps *Dependencies) {
    api := r.Group("/api/v1")

    // 现有路由...

    // 标签管理
    tagHandler := NewTagHandler(deps.TagRepo, deps.AliasRepo)
    api.GET("/tags", tagHandler.GetTags)
    api.POST("/tags", tagHandler.CreateTag)
    api.PUT("/tags/:id", tagHandler.UpdateTag)
    api.DELETE("/tags/:id", tagHandler.DeleteTag)
    api.GET("/tags/:id/aliases", tagHandler.GetAliases)
    api.POST("/tags/:id/aliases", tagHandler.AddAlias)
    api.DELETE("/tags/:id/aliases/:alias_id", tagHandler.DeleteAlias)

    // 图片标签
    imageTagHandler := NewImageTagHandler(deps.ImageTagRepo, deps.TagRepo, deps.GovernanceSvc)
    api.GET("/images/:id/tags", imageTagHandler.GetImageTags)
    api.POST("/images/:id/tags", imageTagHandler.AddImageTag)
    api.DELETE("/images/:id/tags/:tag_id", imageTagHandler.RemoveImageTag)
    api.POST("/images/:id/tags/:tag_id/review", imageTagHandler.ReviewTag)
    api.POST("/images/:id/tags/batch-review", imageTagHandler.BatchReview)

    // AI 标签
    aiTagHandler := NewAITagHandler(deps.JobManager, deps.ImageRepo)
    api.POST("/images/:id/ai-tags", aiTagHandler.TriggerAITags)
    api.GET("/images/:id/ai-tags/status", aiTagHandler.GetAITagStatus)
    api.POST("/images/batch-ai-tags", aiTagHandler.BatchTriggerAITags)
}
```
