type TagRepository interface {
    Save(ctx context.Context, tag *domain.Tag) error
    FindByID(ctx context.Context, id int64) (*domain.Tag, error)
    FindByLabel(ctx context.Context, label string) (*domain.Tag, error)
    FindByLabelLike(ctx context.Context, query string, limit int) ([]*domain.Tag, error)
    FindAll(ctx context.Context, limit, offset int) ([]*domain.Tag, error)
    UpdateReviewState(ctx context.Context, id int64, state string) error
    IncrementUsageCount(ctx context.Context, id int64) error
    Count(ctx context.Context) (int, error)
}

type tagRepository struct {
    db *sql.DB
}

func NewTagRepository(db *sql.DB) TagRepository {
    return &tagRepository{db: db}
}
```

实现要点：
- Save: INSERT OR REPLACE，新标签返回 lastInsertId
- FindByLabel: 精确匹配 preferred_label
- FindByLabelLike: LIKE %query%，按 usage_count DESC 排序
- UpdateReviewState: UPDATE tags SET review_state = ?
- IncrementUsageCount: UPDATE tags SET usage_count = usage_count + 1

type TagGovernanceService struct {
    tagRepo     TagRepository
    obsRepo     TagObservationRepository
    imageTagRepo ImageTagRepository
}

func NewTagGovernanceService(
    tagRepo TagRepository,
    obsRepo TagObservationRepository,
    imageTagRepo ImageTagRepository,
) *TagGovernanceService

// MergeTags 将 AI 标签列表归并到标准标签
// imageID: 图片 ID
// tags: AI 返回的标签列表
// observationID: 来源观测记录 ID
func (s *TagGovernanceService) MergeTags(ctx context.Context, imageID int64, tags []string, observationID int64, confidence float64) error
```

归并逻辑（精确匹配）：
1. 遍历 tags
2. 查询 tagRepo.FindByLabel(tag) 是否存在
3. 存在 → 关联到现有标签，增加使用次数
4. 不存在 → 创建新标签（review_state='pending'），然后关联
5. 创建 image_tags 记录，记录 source_observation_id 和 confidence
