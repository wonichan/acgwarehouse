       ```dart
                               onPressed: () {
                                 _controller.clear();

                             ),

                     ),
                     onSubmitted: (value) {

                         provider.search(value);
                         provider.addToHistory(value);

                     onChanged: (value) => setState(() {}),
                     onTap: () => setState(() => _showHistory = true),
                   ),
                   // 搜索历史下拉

                 ],
               );
             },
           );
         }
         
         Widget _buildHistoryDropdown(SearchProvider provider) {
           return Material(
             elevation: 4,
             child: ListView.builder(
               shrinkWrap: true,
               itemCount: provider.searchHistory.length,
               itemBuilder: (context, index) {
                 final query = provider.searchHistory[index];
                 return ListTile(
                   leading: Icon(Icons.history),
                   title: Text(query),
                   onTap: () {
                     _controller.text = query;
                     provider.search(query);
                     setState(() => _showHistory = false);
                   },
                 );
               },
             ),
           );
         }
       }
       ```

    ```dart
    import 'package:flutter/material.dart';
    import 'package:provider/provider.dart';
    import '../providers/search_provider.dart';
    import '../widgets/search_bar.dart';
    import '../widgets/image_search_upload.dart';
    import '../widgets/image_grid.dart';
    import '../widgets/image_masonry.dart';
    import '../widgets/tag_filter_drawer.dart';
    import '../screens/image_detail_screen.dart';
    
    class SearchScreen extends StatefulWidget {
      @override
      State<SearchScreen> createState() => _SearchScreenState();
    }
    
    class _SearchScreenState extends State<SearchScreen> {
      final ScrollController _scrollController = ScrollController();
      
      @override
      void initState() {
        super.initState();
        context.read<SearchProvider>().loadSearchHistory();
        _scrollController.addListener(_onScroll);
      }
      
      @override
      void dispose() {
        _scrollController.dispose();
        super.dispose();
      }
      
      void _onScroll() {
        if (_scrollController.position.pixels >=
            _scrollController.position.maxScrollExtent - 200) {
          context.read<SearchProvider>().loadMore();
        }
      }
      
      @override
      Widget build(BuildContext context) {
        return Scaffold(
          appBar: AppBar(
            title: _buildAppBarTitle(),
            actions: [
              // 视图切换
              IconButton(
                icon: Icon(context.watch<SearchProvider>().viewMode == ViewMode.grid
                    ? Icons.view_stream
                    : Icons.grid_view),
                onPressed: () => context.read<SearchProvider>().toggleViewMode(),
                tooltip: '切换视图',
              ),
              // 排序
              PopupMenuButton<SortBy>(
                icon: Icon(Icons.sort),
                onSelected: (sortBy) {
                  context.read<SearchProvider>().setSortBy(sortBy);
                  // 重新搜索
                },
                itemBuilder: (context) => [
                  PopupMenuItem(value: SortBy.relevance, child: Text('相关度')),
                  PopupMenuItem(value: SortBy.createdAt, child: Text('时间')),
                  PopupMenuItem(value: SortBy.filename, child: Text('文件名')),
                  PopupMenuItem(value: SortBy.fileSize, child: Text('大小')),
                ],
              ),
              // 标签筛选
              IconButton(
                icon: Icon(Icons.filter_list),
                onPressed: () => _showTagFilter(context),
              ),
            ],
          ),
          body: Consumer<SearchProvider>(
            builder: (context, provider, child) {
              return Column(
                children: [
                  // 搜索模式切换
                  if (provider.searchMode == SearchMode.image)
                    ImageSearchUpload(),
                  
                  // 搜索结果
                  Expanded(child: _buildResults(provider)),
                ],
              );
            },
          ),
        );
      }
      
      Widget _buildAppBarTitle() {
        return SearchBarWidget(
          onImageSearchTap: () {
            context.read<SearchProvider>().setSearchMode(SearchMode.image);
          },
        );
      }
      
      Widget _buildResults(SearchProvider provider) {
        if (provider.isLoading && provider.results.isEmpty) {
          return Center(child: CircularProgressIndicator());
        }
        
        if (provider.errorMessage != null) {
          return Center(
            child: Column(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                Icon(Icons.error, size: 48, color: Colors.red),
                SizedBox(height: 16),
                Text(provider.errorMessage!),
                ElevatedButton(
                  onPressed: () => provider.search(provider.query),
                  child: Text('重试'),
                ),
              ],
            ),
          );
        }
        
        if (provider.results.isEmpty && provider.query.isNotEmpty) {
          return _buildEmptyResult(provider);
        }
        
        if (provider.searchMode == SearchMode.image) {
          return _buildImageSearchResults(provider);
        }
        
        return _buildTextSearchResults(provider);
      }
      
      Widget _buildEmptyResult(SearchProvider provider) {
        return Center(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              Icon(Icons.search_off, size: 64, color: Colors.grey),
              SizedBox(height: 16),
              Text('未找到 "${provider.query}" 的结果'),
              SizedBox(height: 8),
              Text('试试其他关键词或标签'),
            ],
          ),
        );
      }
      
      Widget _buildTextSearchResults(SearchProvider provider) {
        final images = provider.results;
        final viewMode = provider.viewMode;
        
        return RefreshIndicator(
          onRefresh: () => provider.search(provider.query),
          child: viewMode == ViewMode.grid
              ? ImageGrid(
                  images: images,
                  onTap: (image) => _openDetail(image),
                )
              : ImageMasonry(
                  images: images,
                  onTap: (image) => _openDetail(image),
                ),
        );
      }
      
      Widget _buildImageSearchResults(SearchProvider provider) {
        return ListView.builder(
          controller: _scrollController,
          itemCount: provider.imageSearchResults.length,
          itemBuilder: (context, index) {
            final result = provider.imageSearchResults[index];
            return ListTile(
              leading: Image.network(
                result.image.thumbnailUrl,
                width: 64,
                height: 64,
                fit: BoxFit.cover,
              ),
              title: Text(result.image.filename),
              subtitle: Text('相似度: ${(result.similarity * 100).toStringAsFixed(1)}%'),
              trailing: Container(
                width: 60,
                alignment: Alignment.center,

        if (similarity >= 0.5) return Colors.orange;
        return Colors.red;
      }

        Navigator.push(
          context,
          MaterialPageRoute(
            builder: (context) => ImageDetailScreen(image: image),
    ```
