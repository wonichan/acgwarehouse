    ```go
    package e2e_test
    
    import (
        "net/http"
        "net/http/httptest"
        "testing"
        "encoding/json"
        
        "github.com/stretchr/testify/assert"
        "github.com/wonichan/acgwarehouse-backend/internal/handler"
        "github.com/wonichan/acgwarehouse-backend/internal/service"
        "github.com/wonichan/acgwarehouse-backend/internal/repository"
    )
    
    func TestDuplicateDetectionE2E(t *testing.T) {
        // 设置测试数据库
        db := setupTestDB(t)
        defer db.Close()
        
        // 初始化服务和处理器
        imageRepo := repository.NewImageRepository(db)
        duplicateRepo := repository.NewDuplicateRepository(db)
        hashService := service.NewHashService()
        duplicateService := service.NewDuplicateService(imageRepo, duplicateRepo, hashService)
        duplicateHandler := handler.NewDuplicateHandler(duplicateService)
        
        // 设置路由
        router := setupTestRouter()
        api := router.Group("/api/v1")
        duplicateHandler.SetupRoutes(api)
        
        // 测试 1: 触发检测
        t.Run("触发重复检测", func(t *testing.T) {
            req := httptest.NewRequest("POST", "/api/v1/duplicates/detect", 
                strings.NewReader(`{"threshold": 10}`))
            req.Header.Set("Content-Type", "application/json")
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
            var resp map[string]interface{}
            json.NewDecoder(w.Body).Decode(&resp)
            assert.Contains(t, resp, "groups_found")
        })
        
        // 测试 2: 查询重复组
        t.Run("查询重复组列表", func(t *testing.T) {
            req := httptest.NewRequest("GET", "/api/v1/duplicates?limit=10", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
            var resp map[string]interface{}
            json.NewDecoder(w.Body).Decode(&resp)
            assert.Contains(t, resp, "groups")
        })
        
        // 测试 3: 查看组详情
        t.Run("查看重复组详情", func(t *testing.T) {
            req := httptest.NewRequest("GET", "/api/v1/duplicates/1", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
        })
        
        // 测试 4: 删除图片
        t.Run("删除重复图片", func(t *testing.T) {
            req := httptest.NewRequest("DELETE", "/api/v1/duplicates/1", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
        })
    }
    ```

    ```go
    package e2e_test
    
    import (
        "net/http"
        "net/http/httptest"
        "testing"
        "encoding/json"
        "bytes"
        "mime/multipart"
        "os"
        
        "github.com/stretchr/testify/assert"
    )
    
    func TestSearchE2E(t *testing.T) {
        router := setupTestRouter()
        
        // 测试 1: 关键词搜索
        t.Run("关键词搜索", func(t *testing.T) {
            req := httptest.NewRequest("GET", "/api/v1/search?q=test", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
            var resp map[string]interface{}
            json.NewDecoder(w.Body).Decode(&resp)
            assert.Contains(t, resp, "images")
            assert.Contains(t, resp, "total")
        })
        
        // 测试 2: 标签搜索
        t.Run("标签搜索", func(t *testing.T) {
            req := httptest.NewRequest("GET", "/api/v1/search?tag_ids=1,2", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
        })
        
        // 测试 3: 组合搜索
        t.Run("组合搜索", func(t *testing.T) {
            req := httptest.NewRequest("GET", "/api/v1/search?q=cat&tag_ids=1", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
        })
        
        // 测试 4: 排序和分页
        t.Run("排序和分页", func(t *testing.T) {
            req := httptest.NewRequest("GET", 
                "/api/v1/search?q=test&sort_by=created_at&sort_order=desc&limit=10&offset=0", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
        })
        
        // 测试 5: 以图搜图
        t.Run("以图搜图", func(t *testing.T) {
            // 创建 multipart 表单
            body := &bytes.Buffer{}
            writer := multipart.NewWriter(body)
            
            // 添加文件
            file, _ := os.Open("test/fixtures/test_image.jpg")
            defer file.Close()
            part, _ := writer.CreateFormFile("file", "test_image.jpg")
            part.Write([]byte("fake image data"))
            writer.Close()
            
            req := httptest.NewRequest("POST", "/api/v1/search/by-image", body)
            req.Header.Set("Content-Type", writer.FormDataContentType())
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
        })
        
        // 测试 6: 搜索历史
        t.Run("搜索历史", func(t *testing.T) {
            req := httptest.NewRequest("GET", "/api/v1/search/history", nil)
            w := httptest.NewRecorder()
            router.ServeHTTP(w, req)
            
            assert.Equal(t, http.StatusOK, w.Code)
        })
    }
    ```
