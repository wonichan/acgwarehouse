       // SubmitDetection sends POST /compute/duplicates/detect and returns task_id.
       func (c *SidecarClient) SubmitDetection(ctx context.Context, req DetectionRequest) (string, error) {
           body, _ := json.Marshal(req)
           httpReq, _ := http.NewRequestWithContext(ctx, "POST", c.baseURL+"/compute/duplicates/detect", bytes.NewReader(body))
           httpReq.Header.Set("Content-Type", "application/json")
           resp, err := c.httpClient.Do(httpReq)
           if err != nil { return "", fmt.Errorf("sidecar detection request failed: %w", err) }
           defer resp.Body.Close()
           if resp.StatusCode == http.StatusConflict {
               return "", fmt.Errorf("duplicate detection task already running (HTTP 409)")
           }
           if resp.StatusCode != http.StatusOK {
               respBody, _ := io.ReadAll(resp.Body)
               return "", fmt.Errorf("sidecar returned HTTP %d: %s", resp.StatusCode, string(respBody))
           }
           var result DetectionTaskStatus
           json.NewDecoder(resp.Body).Decode(&result)
           return result.TaskID, nil
       }
