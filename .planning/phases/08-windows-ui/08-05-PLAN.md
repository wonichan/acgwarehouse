    import 'package:fluent_ui/fluent_ui.dart';
    import '../models/tag.dart';

    /// Fluent-styled tag list item
    /// Shows tag label, usage count, and action buttons on hover
    class FluentTagListItem extends StatefulWidget {
      final Tag tag;
      final VoidCallback? onTap;
      final VoidCallback? onEdit;
      final VoidCallback? onDelete;
      final bool showActions;

      const FluentTagListItem({
        super.key,
        required this.tag,
        this.onTap,
        this.onEdit,
        this.onDelete,
        this.showActions = true,
      });

      @override
      State<FluentTagListItem> createState() => _FluentTagListItemState();
    }

    class _FluentTagListItemState extends State<FluentTagListItem> {
      bool _isHovered = false;

      @override
      Widget build(BuildContext context) {
        final theme = FluentTheme.of(context);

        return MouseRegion(
          onEnter: (_) => setState(() => _isHovered = true),
          onExit: (_) => setState(() => _isHovered = false),
          cursor: widget.onTap != null ? SystemMouseCursors.click : MouseCursor.defer,
          child: GestureDetector(
            onTap: widget.onTap,
            child: AnimatedContainer(
              duration: const Duration(milliseconds: 100),
              padding: const EdgeInsets.symmetric(horizontal: 12, vertical: 8),
              decoration: BoxDecoration(
                color: _isHovered
                    ? theme.resources.cardBackgroundFillColorSecondary
                    : Colors.transparent,
                borderRadius: BorderRadius.circular(4),
              ),
              child: Row(
                children: [
                  // Tag icon
                  Icon(
                    FluentIcons.tag,
                    size: 16,
                    color: theme.accentColor,
                  ),
                  const SizedBox(width: 12),
                  // Tag label
                  Expanded(
                    child: Column(
                      crossAxisAlignment: CrossAxisAlignment.start,
                      children: [
                        Text(
                          widget.tag.preferredLabel,
                          style: theme.typography.body,
                        ),
                        if (widget.tag.category != null)
                          Text(
                            widget.tag.category!,
                            style: theme.typography.caption?.copyWith(
                              color: theme.resources.textFillColorSecondary,
                            ),
                          ),
                      ],
                    ),
                  ),
                  // Usage count
                  Container(
                    padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 2),
                    decoration: BoxDecoration(
                      color: theme.resources.cardBackgroundFillColorSecondary,
                      borderRadius: BorderRadius.circular(12),
                    ),
                    child: Text(
                      '${widget.tag.usageCount}',
                      style: theme.typography.caption,
                    ),
                  ),
                  // Action buttons (show on hover)
                  if (widget.showActions && _isHovered) ...[
                    const SizedBox(width: 8),
                    if (widget.onEdit != null)
                      Tooltip(
                        message: '编辑',
                        child: IconButton(
                          icon: const Icon(FluentIcons.edit, size: 14),
                          onPressed: widget.onEdit,
                        ),
                      ),
                    if (widget.onDelete != null)
                      Tooltip(
                        message: '删除',
                        child: IconButton(
                          icon: Icon(
                            FluentIcons.delete,
                            size: 14,
                            color: theme.resources.systemFillColorCritical,
                          ),
                          onPressed: widget.onDelete,
                        ),
                      ),
                  ],
                ],
              ),
            ),
          ),
        );
      }
    }
    ```

    /// Fluent-styled tag management page
    /// Contains CommandBar and tag management content
    class FluentTagManagementPage extends StatelessWidget {
      const FluentTagManagementPage({super.key});

      @override
      Widget build(BuildContext context) {
        return Consumer<TagProvider>(
          builder: (context, provider, child) {
            return ScaffoldPage(
              header: PageHeader(
                title: const Text('标签管理'),
                commandBar: CommandBar(
                  mainAxisAlignment: MainAxisAlignment.end,
                  primaryItems: [
                    // Create new tag
                    CommandBarButton(
                      icon: const Icon(FluentIcons.add),
                      label: const Text('新建标签'),
                      onPressed: () => _showCreateDialog(context, provider),
                    ),
                    const CommandBarSeparator(),
                    // Refresh
                    CommandBarButton(
                      icon: const Icon(FluentIcons.refresh),
                      label: const Text('刷新'),
                      onPressed: () => provider.loadTags(),
                    ),
                  ],
                ),
              ),
              content: FluentTagManagementContent(
                onTagTap: (tag) {
                  // Navigate to images with this tag
                  // Set tag filter and navigate to gallery
                  context.read<ImageListProvider>().setTagFilter([tag.id]);
                  context.read<NavigationProvider>().setSelectedIndex(0);
                },
              ),
            );
          },
        );
      }

      void _showCreateDialog(BuildContext context, TagProvider provider) {
        final controller = TextEditingController();
        
        showDialog(
          context: context,
          builder: (context) => ContentDialog(
            title: const Text('新建标签'),
            content: TextBox(
              controller: controller,
              placeholder: '输入标签名称',
              autofocus: true,
            ),
            actions: [
              Button(
                child: const Text('取消'),
                onPressed: () => Navigator.pop(context),
              ),
              FilledButton(
                child: const Text('创建'),
                onPressed: () async {
                  if (controller.text.isNotEmpty) {
                    await provider.createTag(controller.text);
                    if (context.mounted) Navigator.pop(context);
                  }
                },
              ),
            ],
          ),
        );
      }
    }
    ```
