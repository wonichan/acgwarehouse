    import 'package:fluent_ui/fluent_ui.dart';

    /// Gallery page CommandBar with view toggle, sort, and filter actions
    class GalleryCommandBar extends StatelessWidget {
      final VoidCallback? onFilterPressed;
      final VoidCallback? onTagManagementPressed;

      const GalleryCommandBar({
        super.key,
        this.onFilterPressed,
        this.onTagManagementPressed,
      });

      @override
      Widget build(BuildContext context) {
        return Consumer<ImageListProvider>(
          builder: (context, provider, child) {
            return CommandBar(
              mainAxisAlignment: MainAxisAlignment.end,
              primaryItems: [
                // View toggle
                _buildViewToggle(context, provider),
                const CommandBarSeparator(),
                // Sort dropdown
                _buildSortDropdown(context, provider),
                const CommandBarSeparator(),
                // Filter
                CommandBarButton(
                  icon: const Icon(FluentIcons.filter),
                  label: const Text('筛选'),
                  tooltip: const Tooltip(
                    message: '按标签筛选图片',
                    child: Text('筛选'),
                  ),
                  onPressed: onFilterPressed,
                ),
                // Tag management
                CommandBarButton(
                  icon: const Icon(FluentIcons.tag),
                  label: const Text('标签管理'),
                  tooltip: const Tooltip(
                    message: '管理标签（快捷键: Ctrl+T）',
                    child: Text('标签管理'),
                  ),
                  onPressed: onTagManagementPressed ?? () {
                    context.read<NavigationProvider>().setSelectedIndex(3);
                  },
                ),
              ],
            );
          },
        );
      }

      CommandBarBuilderItem _buildViewToggle(
        BuildContext context,
        ImageListProvider provider,
      ) {
        return CommandBarBuilderItem(
          builder: (context, mode, w) => Tooltip(
            message: provider.viewMode == ViewMode.grid
                ? '切换到瀑布流视图'
                : '切换到网格视图',
            child: w,
          ),
          wrappedItem: ToggleButton(
            icon: Icon(
              provider.viewMode == ViewMode.grid
                  ? FluentIcons.bulleted_list_text
                  : FluentIcons.tiles,
            ),
            checked: provider.viewMode == ViewMode.masonry,
            onChanged: (checked) {
              provider.setViewMode(
                checked ? ViewMode.masonry : ViewMode.grid,
              );
            },
          ),
        );
      }

      CommandBarBuilderItem _buildSortDropdown(
        BuildContext context,
        ImageListProvider provider,
      ) {
        return CommandBarBuilderItem(
          builder: (context, mode, w) => const Tooltip(
            message: '选择排序方式',
            child: Text('排序'),
          ),
          wrappedItem: DropDownButton(
            leadingIcon: const Icon(FluentIcons.sort),
            title: const Text('排序'),
            items: [
              MenuFlyoutItem(
                text: const Text('最新导入'),
                onPressed: () => provider.setSort(SortField.createdAt, false),
              ),
              MenuFlyoutItem(
                text: const Text('最早导入'),
                onPressed: () => provider.setSort(SortField.createdAt, true),
              ),
              MenuFlyoutItem(
                text: const Text('文件名 A-Z'),
                onPressed: () => provider.setSort(SortField.filename, true),
              ),
              MenuFlyoutItem(
                text: const Text('文件名 Z-A'),
                onPressed: () => provider.setSort(SortField.filename, false),
              ),
              MenuFlyoutItem(
                text: const Text('文件最大'),
                onPressed: () => provider.setSort(SortField.fileSize, false),
              ),
              MenuFlyoutItem(
                text: const Text('文件最小'),
                onPressed: () => provider.setSort(SortField.fileSize, true),
              ),
            ],
          ),
        );
      }
    }

    /// Search page CommandBar with sort options
    class SearchCommandBar extends StatelessWidget {
      const SearchCommandBar({super.key});

      @override
      Widget build(BuildContext context) {
        return Consumer<SearchProvider>(
          builder: (context, provider, child) {
            return CommandBar(
              mainAxisAlignment: MainAxisAlignment.end,
              primaryItems: [
                // Sort dropdown
                CommandBarBuilderItem(
                  builder: (context, mode, w) => const Tooltip(
                    message: '选择排序方式',
                    child: Text('排序'),
                  ),
                  wrappedItem: DropDownButton(
                    leadingIcon: const Icon(FluentIcons.sort),
                    title: Text(_getSortLabel(provider.sortBy)),
                    items: [
                      MenuFlyoutItem(
                        text: const Text('相关度'),
                        onPressed: () => provider.setSort('relevance', provider.sortOrder),
                      ),
                      MenuFlyoutItem(
                        text: const Text('时间'),
                        onPressed: () => provider.setSort('created_at', provider.sortOrder),
                      ),
                      MenuFlyoutItem(
                        text: const Text('文件名'),
                        onPressed: () => provider.setSort('filename', provider.sortOrder),
                      ),
                      MenuFlyoutItem(
                        text: const Text('大小'),
                        onPressed: () => provider.setSort('file_size', provider.sortOrder),
                      ),
                    ],
                  ),
                ),
                const CommandBarSeparator(),
                // Sort order toggle
                CommandBarButton(
                  icon: Icon(
                    provider.sortOrder == 'desc'
                        ? FluentIcons.sort_lines_descending
                        : FluentIcons.sort_lines_ascending,
                  ),
                  label: Text(provider.sortOrder == 'desc' ? '降序' : '升序'),
                  tooltip: Tooltip(
                    message: provider.sortOrder == 'desc'
                        ? '切换为升序'
                        : '切换为降序',
                    child: Text(provider.sortOrder == 'desc' ? '降序' : '升序'),
                  ),
                  onPressed: () {
                    provider.setSort(
                      provider.sortBy,
                      provider.sortOrder == 'desc' ? 'asc' : 'desc',
                    );
                  },
                ),
              ],
            );
          },
        );
      }

      String _getSortLabel(String sortBy) {
        switch (sortBy) {
          case 'relevance':
            return '相关度';
          case 'created_at':
            return '时间';
          case 'filename':
            return '文件名';
          case 'file_size':
            return '大小';
          default:
            return '排序';
        }
      }
    }
