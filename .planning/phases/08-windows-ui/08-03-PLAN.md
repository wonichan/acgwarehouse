    import 'package:fluent_ui/fluent_ui.dart';
    import '../models/tag.dart';

    /// Tag chip style variants
    enum FluentTagChipStyle {
      confirmed,   // Blue accent
      pending,     // Orange warning
      rejected,    // Grey subdued
    }

    /// Fluent-styled tag chip widget
    class FluentTagChip extends StatelessWidget {
      final Tag tag;
      final FluentTagChipStyle style;
      final VoidCallback? onTap;
      final VoidCallback? onDelete;
      final VoidCallback? onConfirm;
      final VoidCallback? onReject;

      const FluentTagChip({
        super.key,
        required this.tag,
        this.style = FluentTagChipStyle.confirmed,
        this.onTap,
        this.onDelete,
        this.onConfirm,
        this.onReject,
      });

      @override
      Widget build(BuildContext context) {
        final theme = FluentTheme.of(context);
        final colors = _getColors(theme);

        return MouseRegion(
          cursor: onTap != null ? SystemMouseCursors.click : MouseCursor.defer,
          child: GestureDetector(
            onTap: onTap,
            child: Container(
              padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 4),
              decoration: BoxDecoration(
                color: colors.backgroundColor,
                borderRadius: BorderRadius.circular(4),
                border: Border.all(color: colors.borderColor),
              ),
              child: Row(
                mainAxisSize: MainAxisSize.min,
                children: [
                  Text(
                    tag.preferredLabel,
                    style: TextStyle(color: colors.textColor),
                  ),
                  if (style == FluentTagChipStyle.pending) ...[
                    const SizedBox(width: 8),
                    // Confirm button
                    if (onConfirm != null)
                      GestureDetector(
                        onTap: onConfirm,
                        child: Icon(
                          FluentIcons.check_mark,
                          size: 14,
                          color: theme.resources.systemFillColorSuccess,
                        ),
                      ),
                    if (onReject != null) ...[
                      const SizedBox(width: 4),
                      GestureDetector(
                        onTap: onReject,
                        child: Icon(
                          FluentIcons.cancel,
                          size: 14,
                          color: theme.resources.systemFillColorCritical,
                        ),
                      ),
                    ],
                  ],
                  if (onDelete != null && style == FluentTagChipStyle.confirmed) ...[
                    const SizedBox(width: 4),
                    GestureDetector(
                      onTap: onDelete,
                      child: Icon(
                        FluentIcons.chrome_close,
                        size: 12,
                        color: colors.textColor.withOpacity(0.7),
                      ),
                    ),
                  ],
                ],
              ),
            ),
          ),
        );
      }

      _ChipColors _getColors(FluentThemeData theme) {
        switch (style) {
          case FluentTagChipStyle.confirmed:
            return _ChipColors(
              backgroundColor: theme.accentColor.lightest.withOpacity(0.2),
              borderColor: theme.accentColor,
              textColor: theme.accentColor.dark,
            );
          case FluentTagChipStyle.pending:
            return _ChipColors(
              backgroundColor: const Color(0xFFFFF4E5),
              borderColor: const Color(0xFFFFB74D),
              textColor: const Color(0xFFE65100),
            );
          case FluentTagChipStyle.rejected:
            return _ChipColors(
              backgroundColor: theme.resources.cardBackgroundFillColorSecondary,
              borderColor: theme.resources.cardBackgroundFillColorSecondary,
              textColor: theme.resources.textFillColorSecondary,
            );
        }
      }
    }

    class _ChipColors {
      final Color backgroundColor;
      final Color borderColor;
      final Color textColor;

      _ChipColors({
        required this.backgroundColor,
        required this.borderColor,
        required this.textColor,
      });
    }
    ```

    Future<void> showFluentImageDetail(
      BuildContext context, {
      required ImageModel image,
    }) async {
      await showDialog(
        context: context,
        builder: (context) => ContentDialog(
          title: Row(
            children: [
              const Icon(FluentIcons.photo2),
              const SizedBox(width: 8),
              Expanded(
                child: Text(
                  image.filename,
                  overflow: TextOverflow.ellipsis,
                ),
              ),
            ],
          ),
          constraints: const BoxConstraints(
            maxWidth: 800,
            maxHeight: 700,
          ),
          content: FluentImageDetailPanel(image: image),
          actions: [
            Button(
              child: const Text('关闭'),
              onPressed: () => Navigator.pop(context),
            ),
          ],
        ),
      );
    }
    ```
