    ```go
    package service
    
    import (
        "image"
        "os"
        
        "github.com/corona10/goimagehash"
    )
    
    type PHashService struct {
        defaultThreshold int // 默认 12
    }
    
    func NewPHashService() *PHashService {
        return &PHashService{
            defaultThreshold: 12,
        }
    }
    
    // ComputePHash 计算图片的感知哈希
    func (s *PHashService) ComputePHash(imgPath string) (uint64, error) {
        // 1. 打开图片文件
        file, err := os.Open(imgPath)
        if err != nil {
            return 0, err
        }
        defer file.Close()
        
        // 2. 解码图片
        img, _, err := image.Decode(file)
        if err != nil {
            return 0, err
        }
        
        // 3. 计算 pHash
        hash, err := goimagehash.PerceptionHash(img)
        if err != nil {
            return 0, err
        }
        
        return hash.GetHash(), nil
    }
    
    // ComputePHashFromImage 从已解码的 image.Image 计算哈希
    func (s *PHashService) ComputePHashFromImage(img image.Image) (uint64, error) {
        hash, err := goimagehash.PerceptionHash(img)
        if err != nil {
            return 0, err
        }
        return hash.GetHash(), nil
    }
    
    // CompareHashes 比较两个哈希的距离
    // 返回值越小表示越相似，0 表示完全相同
    func (s *PHashService) CompareHashes(hash1, hash2 uint64) (int, error) {
        h1 := goimagehash.NewImageHash(hash1, goimagehash.PHash)
        h2 := goimagehash.NewImageHash(hash2, goimagehash.PHash)
        return h1.Distance(h2)
    }
    
    // IsSimilar 判断两个哈希是否相似
    func (s *PHashService) IsSimilar(hash1, hash2 uint64, threshold int) (bool, error) {
        if threshold <= 0 {
            threshold = s.defaultThreshold
        }
        distance, err := s.CompareHashes(hash1, hash2)
        if err != nil {
            return false, err
        }
        return distance <= threshold, nil
    }
    
    // GetDefaultThreshold 返回默认相似阈值
    func (s *PHashService) GetDefaultThreshold() int {
        return s.defaultThreshold
    }
    ```
    
    使用 `corona10/goimagehash` 库：
    - `goimagehash.PerceptionHash(img)` 计算 pHash
    - `hash.Distance(other)` 计算汉明距离
    
    测试需要准备一些测试图片：
    - 一张原始图片
    - 一张缩放后的相同图片（验证相似性）
    - 一张完全不同的图片（验证不相似性）
