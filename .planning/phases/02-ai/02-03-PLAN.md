    ```go
    package handler
    
    import (
        "net/http"
        "strconv"
        
        "github.com/gin-gonic/gin"
        "github.com/wonichan/acgwarehouse-backend/internal/repository"
    )
    
    type ImagesHandler struct {
        imageRepo *repository.ImageRepository
        jobRepo   *repository.JobRepository
    }
    
    func NewImagesHandler(imageRepo *repository.ImageRepository, jobRepo *repository.JobRepository) *ImagesHandler {
        return &ImagesHandler{
            imageRepo: imageRepo,
            jobRepo:   jobRepo,
        }
    }
    
    // ListImages GET /api/v1/images
    func (h *ImagesHandler) ListImages(c *gin.Context) {
        ctx := c.Request.Context()
        
        // 解析参数
        opts := repository.ListOptions{
            Cursor:  c.Query("cursor"),
            SortBy:  c.Query("sort_by"),
            SortDir: c.Query("sort_dir"),
        }
        
        if limit := c.Query("limit"); limit != "" {
            l, err := strconv.Atoi(limit)
            if err == nil && l > 0 && l <= 100 {
                opts.Limit = l
            }
        }
        
        // 查询
        result, err := h.imageRepo.FindAll(ctx, opts)
        if err != nil {
            c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
            return
        }
        
        c.JSON(http.StatusOK, gin.H{
            "images":      result.Images,
            "next_cursor": result.NextCursor,
            "has_more":    result.HasMore,
        })
    }
    
    // GetImage GET /api/v1/images/:id
    func (h *ImagesHandler) GetImage(c *gin.Context) {
        ctx := c.Request.Context()
        
        idStr := c.Param("id")
        id, err := strconv.ParseInt(idStr, 10, 64)
        if err != nil {
            c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
            return
        }
        
        image, err := h.imageRepo.FindByID(ctx, id)
        if err != nil {
            c.JSON(http.StatusNotFound, gin.H{"error": "image not found"})
            return
        }
        
        c.JSON(http.StatusOK, image)
    }
    
    // GetImportStatus GET /api/v1/images/import-status
    func (h *ImagesHandler) GetImportStatus(c *gin.Context) {
        ctx := c.Request.Context()
        
        // 获取图片总数
        total, err := h.imageRepo.Count(ctx)
        if err != nil {
            c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
            return
        }
        
        // 获取进行中的任务数
        pendingJobs, _ := h.jobRepo.CountByStatus(ctx, "ready")
        runningJobs, _ := h.jobRepo.CountByStatus(ctx, "running")
        
        c.JSON(http.StatusOK, gin.H{
            "total_images":   total,
            "pending_jobs":   pendingJobs,
            "running_jobs":   runningJobs,
            "is_importing":   pendingJobs > 0 || runningJobs > 0,
        })
    }
    ```
    
    API 响应格式参考 CONTEXT.md 中的设计：
    - 列表响应：`{images: [], next_cursor: "", has_more: bool}`
    - 详情响应：完整的 Image 对象
    - 状态响应：`{total_images: N, pending_jobs: N, running_jobs: N, is_importing: bool}`
