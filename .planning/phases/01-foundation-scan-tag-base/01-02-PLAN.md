    2. 更新 cmd/server/main.go：
       - 创建 Gin router
       - 加载配置
       - 注册中间件
       - 注册路由
       - 启动服务器

       ```go
       package main

       import (
           "log"
           
           "github.com/gin-gonic/gin"
           "github.com/yourusername/acgwarehouse-backend/internal/config"
           "github.com/yourusername/acgwarehouse-backend/internal/handler"
           "github.com/yourusername/acgwarehouse-backend/internal/middleware"
       )

       func main() {
           // Load config
           cfg, err := config.LoadConfig()
           if err != nil {
               log.Fatalf("Failed to load config: %v", err)
           }

           // Setup Gin
           if cfg.Server.Env == "production" {
               gin.SetMode(gin.ReleaseMode)
           }
           
           r := gin.New()
           
           // Middleware
           r.Use(middleware.Logger())
           r.Use(middleware.CORS())
           r.Use(gin.Recovery())
           
           // Routes
           handler.SetupRoutes(r)
           
           // Start server
           addr := cfg.Server.Host + ":" + cfg.Server.Port
           log.Printf("ACGWarehouse server starting on %s", addr)
           if err := r.Run(addr); err != nil {
               log.Fatalf("Failed to start server: %v", err)
           }
       }
       ```

       ```go
       // HealthCheck returns the server health status
       func HealthCheck(c *gin.Context) {
           c.JSON(http.StatusOK, gin.H{
               "status":  "ok",
               "message": "ACGWarehouse is running",
               "version": "1.0.0",
           })
       }

       // ReadyCheck returns if the server is ready to accept requests
       func ReadyCheck(c *gin.Context) {
           // TODO: Check database connection
           c.JSON(http.StatusOK, gin.H{
               "status": "ready",
           })
       }
       ```

    2. 创建 internal/handler/routes.go：
       ```go
       package handler

       import (
           "github.com/gin-gonic/gin"
       )

       // SetupRoutes registers all API routes
       func SetupRoutes(r *gin.Engine) {
           // Health endpoints (no version prefix)
           r.GET("/health", HealthCheck)
           r.GET("/ready", ReadyCheck)

           // API v1 routes
           v1 := r.Group("/api/v1")
           {
               // Image routes (placeholder for future)
               images := v1.Group("/images")
               {
                   images.GET("", placeholderHandler)        // List images
                   images.GET("/:id", placeholderHandler)    // Get image
                   images.POST("/scan", placeholderHandler)  // Trigger scan
               }

               // Tag routes (placeholder for future)
               tags := v1.Group("/tags")
               {
                   tags.GET("", placeholderHandler)          // List tags
                   tags.POST("", placeholderHandler)         // Create tag
               }

               // Collection routes (placeholder for future)
               collections := v1.Group("/collections")
               {
                   collections.GET("", placeholderHandler)   // List collections
                   collections.POST("", placeholderHandler)  // Create collection
               }
           }
       }

       // placeholderHandler returns a not implemented response
       func placeholderHandler(c *gin.Context) {
           c.JSON(http.StatusNotImplemented, gin.H{
               "error": "not implemented",
               "hint":  "This endpoint will be implemented in a future phase",
           })
       }
       ```
