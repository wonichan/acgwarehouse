    import (
        "os"

        "github.com/evanoberholster/imagemeta"
        "github.com/yourusername/acgwarehouse-backend/internal/domain"

    type MetadataService struct{}

    func NewMetadataService() *MetadataService {
        return &MetadataService{}
    }

    // SupportedFormats returns supported image formats
    var SupportedFormats = map[string]bool{
        ".jpg":  true,
        ".jpeg": true,
        ".png":  true,
        ".webp": true,
        ".gif":  true,
    }

    // IsImage checks if the file is a supported image format
    func (s *MetadataService) IsImage(path string) bool {
        ext := strings.ToLower(filepath.Ext(path))
        return SupportedFormats[ext]
    }

    // ExtractMetadata extracts metadata from an image file
    func (s *MetadataService) ExtractMetadata(path string) (*domain.Image, error) {
        info, err := os.Stat(path)
        if err != nil {
            return nil, err
        }

        // Open file for metadata extraction
        f, err := os.Open(path)
        if err != nil {
            return nil, err
        }
        defer f.Close()

        // Extract image dimensions using imagemeta
        var width, height int
        imgMeta, err := imagemeta.Decode(f)
        if err == nil {
            width = imgMeta.Width
            height = imgMeta.Height
        }

        // Determine format
        ext := strings.ToLower(filepath.Ext(path))
        format := strings.TrimPrefix(ext, ".")

        return &domain.Image{
            Path:       path,
            Filename:   filepath.Base(path),
            SourceRoot: "", // Will be set by scanner
            FileSize:   info.Size(),
            Width:      width,
            Height:     height,
            Format:     format,
            CreatedAt:  time.Now(),
            UpdatedAt:  time.Now(),
        }, nil
    }

    package service

    import (
        "context"
        "fmt"
        "os"
        "path/filepath"
        "sync"
        "time"
        
        "github.com/yourusername/acgwarehouse-backend/internal/domain"
        "github.com/yourusername/acgwarehouse-backend/internal/repository"
    )

    type ScanResult struct {
        TotalFiles   int
        Imported     int
        Skipped      int
        Failed       int
        Duration     time.Duration
        Errors       []error
    }

    type ScannerService struct {
        metadataSvc *MetadataService
        imageRepo   repository.ImageRepository
        jobRepo     repository.JobRepository
        workers     int
    }

    func NewScannerService(
        metadataSvc *MetadataService,
        imageRepo repository.ImageRepository,
        jobRepo repository.JobRepository,
        workers int,
    ) *ScannerService {
        if workers < 1 {
            workers = 4
        }
        return &ScannerService{
            metadataSvc: metadataSvc,
            imageRepo:   imageRepo,
            jobRepo:     jobRepo,
            workers:     workers,
        }
    }

    // Scan scans the given root directories for images
    func (s *ScannerService) Scan(ctx context.Context, roots []string) (*ScanResult, error) {
        start := time.Now()
        result := &ScanResult{}
        
        // Channel for files to process
        fileChan := make(chan string, 100)
        
        // Wait group for workers
        var wg sync.WaitGroup
        
        // Start workers (second goroutine)
        for i := 0; i < s.workers; i++ {
            wg.Add(1)
            go func() {
                defer wg.Done()
                for path := range fileChan {
                    if err := s.processFile(path, roots); err != nil {
                        result.Failed++
                        result.Errors = append(result.Errors, 
                            fmt.Errorf("failed to process %s: %w", path, err))
                    } else {
                        result.Imported++
                    }
                }
            }()
        }
        
        // Walk filesystem (first goroutine)
        for _, root := range roots {
            filepath.Walk(root, func(path string, info os.FileInfo, err error) error {
                if err != nil {
                    result.Errors = append(result.Errors, err)
                    return nil
                }
                if info.IsDir() {
                    return nil
                }
                if !s.metadataSvc.IsImage(path) {
                    result.Skipped++
                    return nil
                }
                result.TotalFiles++
                fileChan <- path
                return nil
            })
        }
        
        close(fileChan)
        wg.Wait()
        
        result.Duration = time.Since(start)
        return result, nil
    }

    func (s *ScannerService) processFile(path string, roots []string) error {
        // Extract metadata
        image, err := s.metadataSvc.ExtractMetadata(path)
        if err != nil {
            return err
        }
        
        // Find source root
        for _, root := range roots {
            if filepath.Dir(path) == root || 
               len(path) > len(root) && path[:len(root)] == root {
                image.SourceRoot = root
                break
            }
        }
        
        // Save to database
        if err := s.imageRepo.Save(image); err != nil {
            return err
        }
        
        // Queue "image_imported" job for AI tagging
        if s.jobRepo != nil {
            job := &domain.AsyncJob{
                Type:    "image_imported",
                Status:  "ready",
                Payload: fmt.Sprintf(`{"image_id": %d, "path": "%s"}`, image.ID, image.Path),
            }
            s.jobRepo.Save(job)
        }
        
        return nil
    }
    ```

        result, err := scannerSvc.Scan(nil, paths)
        if err != nil {
            log.Fatalf("Scan failed: %v", err)

       package service

       import (
           "context"
           "log"
           "path/filepath"
           "sync"
           "time"
           
           "github.com/fsnotify/fsnotify"
           "github.com/yourusername/acgwarehouse-backend/internal/domain"
           "github.com/yourusername/acgwarehouse-backend/internal/repository"
       )

       type WatcherService struct {
           watcher      *fsnotify.Watcher
           scannerSvc   *ScannerService
           metadataSvc  *MetadataService
           imageRepo    repository.ImageRepository
           jobRepo      repository.JobRepository
           roots        []string
           debounceTime time.Duration
           mu           sync.Mutex
           pending      map[string]time.Time
           stopChan     chan struct{}
       }

       func NewWatcherService(
           scannerSvc *ScannerService,
           metadataSvc *MetadataService,
           imageRepo repository.ImageRepository,
           jobRepo repository.JobRepository,
           roots []string,
       ) (*WatcherService, error) {
           watcher, err := fsnotify.NewWatcher()
           if err != nil {
               return nil, err
           }
           
           return &WatcherService{
               watcher:      watcher,
               scannerSvc:   scannerSvc,
               metadataSvc:  metadataSvc,
               imageRepo:    imageRepo,
               jobRepo:      jobRepo,
               roots:        roots,
               debounceTime: 500 * time.Millisecond,
               pending:      make(map[string]time.Time),
               stopChan:     make(chan struct{}),
           }, nil
       }

       // Start begins watching the configured directories
       func (w *WatcherService) Start(ctx context.Context) error {
           // Add all roots to watcher
           for _, root := range w.roots {
               if err := w.watcher.Add(root); err != nil {
                   log.Printf("Warning: could not watch %s: %v", root, err)
               }
               log.Printf("Watching: %s", root)
           }
           
           // Start debounced processor
           go w.processDebounced()
           
           // Process events
           for {
               select {
               case <-ctx.Done():
                   return ctx.Err()
               case <-w.stopChan:
                   return nil
               case event, ok := <-w.watcher.Events:
                   if !ok {
                       return nil
                   }
                   w.handleEvent(event)
               case err, ok := <-w.watcher.Errors:
                   if !ok {
                       return nil
                   }
                   log.Printf("Watcher error: %v", err)
               }
           }
       }

       func (w *WatcherService) handleEvent(event fsnotify.Event) {
           // Only handle create events
           if event.Op&fsnotify.Create != fsnotify.Create {
               return
           }
           
           // Check if it's an image
           if !w.metadataSvc.IsImage(event.Name) {
               return
           }
           
           // Add to pending with debounce
           w.mu.Lock()
           w.pending[event.Name] = time.Now()
           w.mu.Unlock()
       }

       func (w *WatcherService) processDebounced() {
           ticker := time.NewTicker(w.debounceTime)
           defer ticker.Stop()
           
           for {
               select {
               case <-w.stopChan:
                   return
               case <-ticker.C:
                   w.processPending()
               }
           }
       }

       func (w *WatcherService) processPending() {
           w.mu.Lock()
           defer w.mu.Unlock()
           
           now := time.Now()
           for path, added := range w.pending {
               if now.Sub(added) >= w.debounceTime {
                   go w.importFile(path)
                   delete(w.pending, path)
               }
           }
       }

       func (w *WatcherService) importFile(path string) {
           log.Printf("New image detected: %s", path)
           
           // Extract metadata
           image, err := w.metadataSvc.ExtractMetadata(path)
           if err != nil {
               log.Printf("Failed to extract metadata for %s: %v", path, err)
               return
           }
           
           // Find source root
           for _, root := range w.roots {
               absPath, _ := filepath.Abs(path)
               absRoot, _ := filepath.Abs(root)
               if len(absPath) >= len(absRoot) && absPath[:len(absRoot)] == absRoot {
                   image.SourceRoot = root
                   break
               }
           }
           
           // Save to database
           if err := w.imageRepo.Save(image); err != nil {
               log.Printf("Failed to save image %s: %v", path, err)
               return
           }
           
           // Queue job for AI tagging
           job := &domain.AsyncJob{
               Type:    "image_imported",
               Status:  "ready",
               Payload: fmt.Sprintf(`{"image_id": %d, "path": "%s"}`, image.ID, image.Path),
           }
           w.jobRepo.Save(job)
           
           log.Printf("Imported: %s (ID: %d)", path, image.ID)
       }

       // Stop stops the watcher
       func (w *WatcherService) Stop() {
           close(w.stopChan)
           w.watcher.Close()
       }
       ```

    package repository

    import (
        "database/sql"
        "encoding/json"
        
        "github.com/yourusername/acgwarehouse-backend/internal/domain"
    )

    type JobRepository interface {
        Save(job *domain.AsyncJob) error
        FindByID(id int64) (*domain.AsyncJob, error)
        FindByStatus(status string) ([]domain.AsyncJob, error)
        Update(job *domain.AsyncJob) error
    }

    type sqliteJobRepository struct {
        db *sql.DB
    }

    func NewJobRepository(db *sql.DB) JobRepository {
        return &sqliteJobRepository{db: db}
    }

    func (r *sqliteJobRepository) Save(job *domain.AsyncJob) error {
        result, err := r.db.Exec(`
            INSERT INTO async_jobs (type, status, payload, progress, error, created_at)
            VALUES (?, ?, ?, ?, ?, ?)
        `, job.Type, job.Status, job.Payload, job.Progress, job.Error, job.CreatedAt)
        if err != nil {
            return err
        }
        id, _ := result.LastInsertId()
        job.ID = id
        return nil
    }

    func (r *sqliteJobRepository) FindByID(id int64) (*domain.AsyncJob, error) {
        job := &domain.AsyncJob{}
        err := r.db.QueryRow(`
            SELECT id, type, status, payload, progress, error, created_at, started_at, finished_at
            FROM async_jobs WHERE id = ?
        `, id).Scan(&job.ID, &job.Type, &job.Status, &job.Payload, &job.Progress,
             &job.Error, &job.CreatedAt, &job.StartedAt, &job.FinishedAt)
        return job, err
    }

    func (r *sqliteJobRepository) FindByStatus(status string) ([]domain.AsyncJob, error) {
        rows, err := r.db.Query(`
            SELECT id, type, status, payload, progress, error, created_at, started_at, finished_at
            FROM async_jobs WHERE status = ? ORDER BY created_at
        `, status)
        if err != nil {
            return nil, err
        }
        defer rows.Close()
        
        var jobs []domain.AsyncJob
        for rows.Next() {
            job := domain.AsyncJob{}
            rows.Scan(&job.ID, &job.Type, &job.Status, &job.Payload, &job.Progress,
                      &job.Error, &job.CreatedAt, &job.StartedAt, &job.FinishedAt)
            jobs = append(jobs, job)
        }
        return jobs, nil
    }

    func (r *sqliteJobRepository) Update(job *domain.AsyncJob) error {
        _, err := r.db.Exec(`
            UPDATE async_jobs SET status = ?, progress = ?, error = ?, started_at = ?, finished_at = ?
            WHERE id = ?
        `, job.Status, job.Progress, job.Error, job.StartedAt, job.FinishedAt, job.ID)
        return err
    }

    package worker

    import (
        "context"
        "log"
        "sync"
        "time"
        
        "github.com/yourusername/acgwarehouse-backend/internal/domain"
        "github.com/yourusername/acgwarehouse-backend/internal/repository"
    )

    // JobFunc is a function that processes a job
    type JobFunc func(ctx context.Context, job *domain.AsyncJob) error

    // Manager manages async jobs
    // Reference: PHASE1_PATTERNS.md - stashapp/stash job manager pattern
    type Manager struct {
        jobRepo   repository.JobRepository
        handlers  map[string]JobFunc
        queue     []*domain.AsyncJob
        mu        sync.Mutex
        notEmpty  *sync.Cond
        stopChan  chan struct{}
        running   bool
    }

    func NewManager(jobRepo repository.JobRepository) *Manager {
        m := &Manager{
            jobRepo:  jobRepo,
            handlers: make(map[string]JobFunc),
            queue:    make([]*domain.AsyncJob, 0),
            stopChan: make(chan struct{}),
        }
        m.notEmpty = sync.NewCond(&m.mu)
        return m
    }

    // RegisterHandler registers a handler for a job type
    func (m *Manager) RegisterHandler(jobType string, handler JobFunc) {
        m.handlers[jobType] = handler
    }

    // Start starts the job processor
    func (m *Manager) Start(ctx context.Context) {
        m.mu.Lock()
        m.running = true
        m.mu.Unlock()
        
        go m.dispatcher(ctx)
        go m.pollDatabase(ctx)
        
        log.Println("Job manager started")
    }

    // Stop stops the job processor
    func (m *Manager) Stop() {
        m.mu.Lock()
        m.running = false
        m.notEmpty.Broadcast()
        m.mu.Unlock()
        
        close(m.stopChan)
        log.Println("Job manager stopped")
    }

    // pollDatabase periodically checks for ready jobs
    func (m *Manager) pollDatabase(ctx context.Context) {
        ticker := time.NewTicker(5 * time.Second)
        defer ticker.Stop()
        
        for {
            select {
            case <-ctx.Done():
                return
            case <-m.stopChan:
                return
            case <-ticker.C:
                m.loadReadyJobs()
            }
        }
    }

    func (m *Manager) loadReadyJobs() {
        jobs, err := m.jobRepo.FindByStatus("ready")
        if err != nil {
            log.Printf("Failed to load ready jobs: %v", err)
            return
        }
        
        m.mu.Lock()
        defer m.mu.Unlock()
        
        for _, job := range jobs {
            // Check if already in queue
            found := false
            for _, qj := range m.queue {
                if qj.ID == job.ID {
                    found = true
                    break
                }
            }
            if !found {
                m.queue = append(m.queue, job)
                m.notEmpty.Broadcast()
            }
        }
    }

    // dispatcher processes jobs one at a time
    func (m *Manager) dispatcher(ctx context.Context) {
        for {
            m.mu.Lock()
            
            // Wait for a job
            for len(m.queue) == 0 && m.running {
                m.notEmpty.Wait()
            }
            
            if !m.running {
                m.mu.Unlock()
                return
            }
            
            // Get next job
            job := m.queue[0]
            m.queue = m.queue[1:]
            m.mu.Unlock()
            
            // Process job
            m.processJob(ctx, job)
        }
    }

    func (m *Manager) processJob(ctx context.Context, job *domain.AsyncJob) {
        handler, ok := m.handlers[job.Type]
        if !ok {
            log.Printf("No handler for job type: %s", job.Type)
            job.Status = "failed"
            job.Error = strPtr("no handler registered")
            m.jobRepo.Update(job)
            return
        }
        
        // Mark as running
        now := time.Now()
        job.Status = "running"
        job.StartedAt = &now
        m.jobRepo.Update(job)
        
        // Execute handler
        if err := handler(ctx, job); err != nil {
            log.Printf("Job %d failed: %v", job.ID, err)
            job.Status = "failed"
            job.Error = strPtr(err.Error())
        } else {
            job.Status = "finished"
        }
        
        // Mark as done
        finished := time.Now()
        job.FinishedAt = &finished
        m.jobRepo.Update(job)
        
        log.Printf("Job %d (%s) %s", job.ID, job.Type, job.Status)
    }

    func strPtr(s string) *string {
        return &s
    }
