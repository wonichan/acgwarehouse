  <files>flutter_app/lib/app/fluent_app_shell.dart</files>
  <action>
    创建 flutter_app/lib/app/fluent_app_shell.dart:
    
    ```dart
    import 'package:fluent_ui/fluent_ui.dart';
    import 'package:provider/provider.dart';
    
    import '../providers/navigation_provider.dart';
    import 'fluent_screens.dart';
    
    /// FluentApp Shell - Windows 桌面端导航框架
    /// 包含 NavigationView 侧边导航栏和页面容器
    class FluentAppShell extends StatelessWidget {
      const FluentAppShell({super.key});
      
      @override
      Widget build(BuildContext context) {
        return Consumer<NavigationProvider>(
          builder: (context, navProvider, child) {
            return NavigationView(
              appBar: const NavigationAppBar(
                title: Text('ACGWarehouse'),
                // Phase 8 完整实现窗口控制按钮
              ),
              pane: NavigationPane(
                selected: navProvider.selectedIndex,
                onChanged: (index) {
                  navProvider.setSelectedIndex(index);
                },
                displayMode: PaneDisplayMode.auto,
                items: const [
                  PaneItem(
                    icon: Icon(FluentIcons.photo_library),
                    title: Text('图库'),
                  ),
                  PaneItem(
                    icon: Icon(FluentIcons.search),
                    title: Text('搜索'),
                  ),
                  PaneItem(
                    icon: Icon(FluentIcons.copy),
                    title: Text('重复检测'),
                  ),
                ],
              ),
              content: NavigationBody(
                index: navProvider.selectedIndex,
                children: const [
                  FluentGalleryPage(),
                  FluentSearchPage(),
                  FluentDuplicatePage(),
                ],
              ),
            );
          },
        );
      }
    }
    ```
    
    注意：
    - 使用 Consumer<NavigationProvider> 监听导航状态变化
    - PaneDisplayMode.auto 自动适应窗口大小
    - 导航项与 MaterialApp 保持一致（图库、搜索、重复检测）
    - content 使用 NavigationBody 切换页面
  </action>
